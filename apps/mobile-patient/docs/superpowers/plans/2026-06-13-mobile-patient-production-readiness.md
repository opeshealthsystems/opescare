```dart

```
Replace with:
```dart
// Injected at build time: --dart-define=API_BASE_URL=https://mobile-api.opescare.com/api
static const String _base =
    String.fromEnvironment('API_BASE_URL', defaultValue: 'http://opescare.test/api');

static String get baseUrl {
  // In release builds, refuse to fall back to the local HTTP dev default.
  const bool isRelease = bool.fromEnvironment('dart.vm.product');
  if (isRelease && (_base == 'http://opescare.test/api' || _base.startsWith('http://'))) {
    throw StateError(
      'Production build requires an HTTPS API_BASE_URL via --dart-define=API_BASE_URL=...',
    );
  }
  return _base;
}
```

```dart
import 'package:local_auth/local_auth.dart';

class AppLockService {
  final _auth = LocalAuthentication();
  DateTime? _lastActive;
  static const lockAfter = Duration(minutes: 2);

  void markActive() => _lastActive = DateTime.now();

  bool get shouldLock {
    if (_lastActive == null) return true;
    return DateTime.now().difference(_lastActive!) > lockAfter;
  }

  Future<bool> authenticate() async {
    final canBio = await _auth.canCheckBiometrics;
    if (canBio) {
      return _auth.authenticate(
        localizedReason: 'Unlock OpesCare',
        options: const AuthenticationOptions(stickyAuth: true, biometricOnly: false),
      );
    }
    return false; // fall through to PIN screen
  }
}
```

```bash
echo | openssl s_client -servername api.opescare.com -connect api.opescare.com:443 2>/dev/null \
  | openssl x509 -pubkey -noout \
  | openssl pkey -pubin -outform der \
```

Record the leaf + an intermediate/backup pin.

- [ ] **Step 2: Attach a pinning validator to the dio `HttpClientAdapter`**

```dart
import 'dart:io';
import 'package:dio/io.dart';

const _pins = <String>{ 'BASE64_LEAF_PIN', 'BASE64_BACKUP_PIN' };

void applyCertPinning(Dio dio) {
  (dio.httpClientAdapter as IOHttpClientAdapter).createHttpClient = () {
    final client = HttpClient();
    client.badCertificateCallback = (cert, host, port) {
      final spkiSha256 = _spkiSha256Base64(cert); // compute from cert.der
      return _pins.contains(spkiSha256);
    };
    return client;
  };
}
```
